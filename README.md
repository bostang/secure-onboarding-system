```bash
# adding submodules of project:
git submodule add https://github.com/bostang/verificator-secure-onboarding-system
git submodule add https://github.com/bostang/backend-secure-onboarding-system
git submodule add https://github.com/alvarolt17/frontend-secure-onboarding-system

# pull down the contents of the submodules
git submodule update --init --recursive

```
